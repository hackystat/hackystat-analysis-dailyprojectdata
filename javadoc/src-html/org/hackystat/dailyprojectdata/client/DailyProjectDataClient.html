001    package org.hackystat.dailyprojectdata.client;
002    
003    import java.io.StringReader;
004    
005    import javax.xml.bind.JAXBContext;
006    import javax.xml.bind.Unmarshaller;
007    import javax.xml.datatype.XMLGregorianCalendar;
008    
009    import org.hackystat.dailyprojectdata.resource.devtime.jaxb.DevTimeDailyProjectData;
010    import org.hackystat.utilities.tstamp.Tstamp;
011    import org.restlet.Client;
012    import org.restlet.data.ChallengeResponse;
013    import org.restlet.data.ChallengeScheme;
014    import org.restlet.data.MediaType;
015    import org.restlet.data.Method;
016    import org.restlet.data.Preference;
017    import org.restlet.data.Protocol;
018    import org.restlet.data.Reference;
019    import org.restlet.data.Request;
020    import org.restlet.data.Response;
021    import org.restlet.data.Status;
022    import org.restlet.resource.Representation;
023    
024    /**
025     * Provides a client to support access to the DailyProjectData service. 
026     * @author Philip Johnson
027     */
028    public class DailyProjectDataClient {
029      
030      /** Holds the userEmail to be associated with this client. */
031      private String userEmail;
032      /** Holds the password to be associated with this client. */
033      private String password;
034      /** The DailyProjectData host, such as "http://localhost:9877/dailyprojectdata". */
035      private String dailyProjectDataHost;
036      /** The Restlet Client instance used to communicate with the server. */
037      private Client client;
038      /** DevTime JAXBContext */
039      private JAXBContext devTimeJAXB;
040      /** The http authentication approach. */
041      private ChallengeScheme scheme = ChallengeScheme.HTTP_BASIC;
042      /** The preferred representation type. */
043      private Preference<MediaType> xmlMedia = new Preference<MediaType>(MediaType.TEXT_XML);
044      /** To facilitate debugging of problems using this system. */
045      private boolean isTraceEnabled = false;
046      
047      /**
048       * Initializes a new DailyProjectDataClient, given the host, userEmail, and password. 
049       * Note that the userEmail and password refer to the underlying SensorBase client
050       * associated with this DailyProjectData service.  This service does not keep its own
051       * independent set of userEmails and passwords.  Authentication is not actually performed
052       * in this constructor. Use the authenticate() method to explicitly check the authentication
053       * credentials. 
054       * @param host The host, such as 'http://localhost:9877/dailyprojectdata'.
055       * @param email The user's email used for authentication. 
056       * @param password The password used for authentication.
057       */
058      public DailyProjectDataClient(String host, String email, String password) {
059        validateArg(host);
060        validateArg(email);
061        validateArg(password);
062        this.userEmail = email;
063        this.password = password;
064        this.dailyProjectDataHost = host;
065        if (!this.dailyProjectDataHost.endsWith("/")) {
066          this.dailyProjectDataHost = this.dailyProjectDataHost + "/";
067        }
068        if (this.isTraceEnabled) {
069          System.out.println("DailyProjectDataClient Tracing: INITIALIZE " + 
070              "host='" + host + "', email='" + email + "', password='" + password + "'");
071        }
072        this.client = new Client(Protocol.HTTP);
073        try {
074          this.devTimeJAXB = 
075            JAXBContext.newInstance(
076                org.hackystat.dailyprojectdata.resource.devtime.jaxb.ObjectFactory.class);
077        }
078        catch (Exception e) {
079          throw new RuntimeException("Couldn't create JAXB context instances.", e);
080        }
081      }
082      
083      /**
084       * Throws an unchecked illegal argument exception if the arg is null or empty. 
085       * @param arg The String that must be non-null and non-empty. 
086       */
087      private void validateArg(String arg) {
088        if ((arg == null) || ("".equals(arg))) {
089          throw new IllegalArgumentException(arg + " cannot be null or the empty string.");
090        }
091      }
092      
093      /**
094       * Does the housekeeping for making HTTP requests to the SensorBase by a test or admin user. 
095       * @param method The type of Method.
096       * @param requestString A string, such as "users". No preceding slash. 
097       * @param entity The representation to be sent with the request, or null if not needed.  
098       * @return The Response instance returned from the server.
099       */
100      private Response makeRequest(Method method, String requestString, Representation entity) {
101        Reference reference = new Reference(this.dailyProjectDataHost + requestString);
102        Request request = (entity == null) ? 
103            new Request(method, reference) :
104              new Request(method, reference, entity);
105        request.getClientInfo().getAcceptedMediaTypes().add(xmlMedia); 
106        ChallengeResponse authentication = new ChallengeResponse(scheme, this.userEmail, this.password);
107        request.setChallengeResponse(authentication);
108        if (this.isTraceEnabled) {
109          System.out.println("SensorBaseClient Tracing: " + method + " " + reference);
110          if (entity != null) {
111            try {
112              System.out.println(entity.getText());
113            }
114            catch (Exception e) {
115              System.out.println("  Problems with getText() on entity.");
116            }
117          }
118        }
119        Response response = this.client.handle(request);
120        if (this.isTraceEnabled) {
121          Status status = response.getStatus();
122          System.out.println("  => " + status.getCode() + " " + status.getDescription());
123        }
124        return response;
125      }
126      
127      /**
128       * Takes a String encoding of a DevTimeDailyProjectData in XML format and converts it. 
129       * @param xmlString The XML string representing a DevTimeDailyProjectData.
130       * @return The corresponding DevTimeDailyProjectData instance. 
131       * @throws Exception If problems occur during unmarshalling.
132       */
133      private DevTimeDailyProjectData makeDevTimeDailyProjectData(String xmlString) throws Exception {
134        Unmarshaller unmarshaller = this.devTimeJAXB.createUnmarshaller();
135        return (DevTimeDailyProjectData)unmarshaller.unmarshal(new StringReader(xmlString));
136      }
137      
138      /**
139       * Authenticates this user and password with this DailyProjectData service, throwing a
140       * DailyProjectDataException if the user and password associated with this instance
141       * are not valid credentials. 
142       * Note that authentication is performed by checking these credentials with the 
143       * SensorBase; this service does not keep its own independent set of usernames and passwords.
144       * @return This DailyProjectDataClient instance. 
145       * @throws DailyProjectDataClientException If authentication is not successful. 
146       */
147      public synchronized DailyProjectDataClient authenticate() throws DailyProjectDataClientException {
148        // Performs authentication by running the DevTime analysis for the next 24 hours, which is 
149        // likely to involve little to no data and thus be quite cheap. We could convert to a 'ping'
150        // style interface in future if we need to. 
151        String uri = "devtime/" + this.userEmail + "/" + "Default/" + Tstamp.makeTimestamp(); 
152        Response response = makeRequest(Method.GET, uri, null); 
153        if (!response.getStatus().isSuccess()) {
154          throw new DailyProjectDataClientException(response.getStatus());
155        }
156        return this;
157      }
158      
159      /**
160       * Returns a DevTimeDailyProjectData instance from this server, or throws a
161       * DailyProjectData exception if problems occurred.  
162       * @param user The user. 
163       * @param project The project.
164       * @param timestamp The Timestamp indicating the start of the 24 hour period of DevTime.
165       * @return The DevTimeDailyProjectData instance. 
166       * @throws DailyProjectDataClientException If the credentials associated with this instance
167       * are not valid, or if the underlying SensorBase service cannot be reached, or if one or more
168       * of the supplied user, password, or timestamp is not valid.
169       */
170      public synchronized DevTimeDailyProjectData getDevTime(String user, String project, 
171          XMLGregorianCalendar timestamp) throws DailyProjectDataClientException {
172        Response response = makeRequest(Method.GET, "devtime/" + user + "/" + project + "/" + timestamp,
173            null);
174        DevTimeDailyProjectData devTime;
175        if (!response.getStatus().isSuccess()) {
176          throw new DailyProjectDataClientException(response.getStatus());
177        }
178        try {
179          String xmlData = response.getEntity().getText();
180          devTime = makeDevTimeDailyProjectData(xmlData);
181        }
182        catch (Exception e) {
183          throw new DailyProjectDataClientException(response.getStatus(), e);
184        }
185        return devTime;
186      } 
187      
188    }
