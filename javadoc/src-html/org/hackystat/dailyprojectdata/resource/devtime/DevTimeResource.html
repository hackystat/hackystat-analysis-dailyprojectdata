031    /**
032     * Implements the Resource for processing GET {host}/devtime/{user}/{project}/{starttime} requests.
033     * Requires the authenticated user to be {user} or else the Admin user for the sensorbase 
034     * connected to this service. 
035     * @author Philip Johnson
036     */
037    public class DevTimeResource extends DailyProjectDataResource {
038      
039      /**
040       * The standard constructor.
041       * @param context The context.
042       * @param request The request object.
043       * @param response The response object.
044       */
045      public DevTimeResource(Context context, Request request, Response response) {
046        super(context, request, response);
047      }
048      
049      /**
050       * Returns an DevTimeDailyProjectData instance representing the DevTime associated with the 
051       * Project data, or null if not authorized. 
052       * @param variant The representational variant requested.
053       * @return The representation. 
054       */
055      @Override
056      public Representation getRepresentation(Variant variant) {
057        if (variant.getMediaType().equals(MediaType.TEXT_XML)) {
058          try {
059            // [1] get the SensorBaseClient for the user making this request.
060            SensorBaseClient client = super.getSensorBaseClient();
061            // [2] get a SensorDataIndex of all sensor data for this Project on the requested day.
062            XMLGregorianCalendar startTime = Tstamp.makeTimestamp(this.timestamp);
063            XMLGregorianCalendar endTime = Tstamp.incrementDays(startTime, 1);
064            SensorDataIndex index = client.getProjectSensorData(authUser, project, startTime, endTime);
065            // [3] look through this index for DevEvent sensor data, and update the DevTime counter. 
066            MemberDevTimeCounter counter = new MemberDevTimeCounter();
067            for (SensorDataRef ref : index.getSensorDataRef()) {
068              if (ref.getSensorDataType().equals("DevEvent")) {
069                // If it's a devEvent, get the member and timestamp and update the MemberDevTimeCounter.
070                counter.addMemberDevEvent(ref.getOwner(), ref.getTimestamp());
071              }
072            }
073            // [4] create and return the DevTimeDailyProjectData
074            DevTimeDailyProjectData devTime = new DevTimeDailyProjectData();
075            //     create the individuam MemberData elements. 
076            for (String member : counter.getMembers()) {
077              MemberData memberData = new MemberData();
078              memberData.setMemberUri(member);
079              memberData.setDevTime(counter.getMemberDevTime(member));
080              devTime.getMemberData().add(memberData);
081            }
082            devTime.setOwner(uriUser);
083            devTime.setProject(project);
084            devTime.setUriPattern("**"); // we don't support UriPatterns yet. 
085            devTime.setTotalDevTime(counter.getTotalDevTime());
086            String xmlData = makeDevTime(devTime);
087            return super.getStringRepresentation(xmlData);
088          }
089          catch (Exception e) {
090            server.getLogger().warning("Error processing devTime: " + StackTrace.toString(e));
091            return null;
092          }
093        }
094        return null;
095      }
096      
097      /**
098       * Returns the passed SensorData instance as a String encoding of its XML representation.
099       * @param data The SensorData instance. 
100       * @return The XML String representation.
101       * @throws Exception If problems occur during translation. 
102       */
103      private String makeDevTime (DevTimeDailyProjectData data) throws Exception {
104        JAXBContext devTimeJAXB = 
105          (JAXBContext)this.server.getContext().getAttributes().get("DevTimeJAXB");
106        Marshaller marshaller = devTimeJAXB.createMarshaller(); 
107        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
108        dbf.setNamespaceAware(true);
109        DocumentBuilder documentBuilder = dbf.newDocumentBuilder();
110        Document doc = documentBuilder.newDocument();
111        marshaller.marshal(data, doc);
112        DOMSource domSource = new DOMSource(doc);
113        StringWriter writer = new StringWriter();
114        StreamResult result = new StreamResult(writer);
115        TransformerFactory tf = TransformerFactory.newInstance();
116        Transformer transformer = tf.newTransformer();
117        transformer.transform(domSource, result);
118        return writer.toString();
119      }
120    }
