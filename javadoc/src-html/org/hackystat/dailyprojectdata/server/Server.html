028    /**
029     * Sets up the HTTP Server process and dispatching to the associated resources. 
030     * @author Philip Johnson
031     */
032    public class Server extends Application { 
033    
034      /** Holds the Restlet Component associated with this Server. */
035      private Component component; 
036      
037      /** Holds the host name associated with this Server. */
038      private String hostName;
039      
040      /** Holds the HackystatLogger for this Service. */
041      private Logger logger; 
042      
043      /** Holds the ServerProperties instance for this Service. */
044      private ServerProperties properties;
045      
046      /**
047       * Creates a new instance of a DailyProjectData HTTP server, listening on the supplied port.  
048       * @return The Server instance created. 
049       * @throws Exception If problems occur starting up this server. 
050       */
051      public static Server newInstance() throws Exception {
052        Server server = new Server();
053        server.logger = HackystatLogger.getLogger("org.hackystat.dailyprojectdata");
054        server.properties = new ServerProperties();
055        server.hostName = "http://" +
056                          server.properties.get(HOSTNAME_KEY) + 
057                          ":" + 
058                          server.properties.get(PORT_KEY) + 
059                          "/" +
060                          server.properties.get(CONTEXT_ROOT_KEY) +
061                          "/";
062        int port = Integer.valueOf(server.properties.get(PORT_KEY));
063        server.component = new Component();
064        server.component.getServers().add(Protocol.HTTP, port);
065        server.component.getDefaultHost()
066          .attach("/" + server.properties.get(CONTEXT_ROOT_KEY), server);
067     
068        
069        // Create and store the JAXBContext instances on the server context.
070        // They are supposed to be thread safe. 
071        Map<String, Object> attributes = server.getContext().getAttributes();
072        JAXBContext devTimeJAXB = JAXBContext.newInstance(
073            org.hackystat.dailyprojectdata.resource.devtime.jaxb.ObjectFactory.class);
074        attributes.put("DevTimeJAXB", devTimeJAXB);
075        
076        // Provide a pointer to this server in the Context so that Resources can get at this server.
077        attributes.put("DailyProjectDataServer", server);
078        
079        // Now let's open for business. 
080        server.logger.warning("Host: " + server.hostName);
081        HackystatLogger.setLoggingLevel(server.logger, server.properties.get(LOGGING_LEVEL_KEY));
082        server.properties.echoProperties(server);
083        String sensorBaseHost = server.properties.get(SENSORBASE_HOST_KEY);
084        boolean sensorBaseOK = SensorBaseClient.isHost(sensorBaseHost);
085        server.logger.warning("SensorBase " + sensorBaseHost + 
086            ((sensorBaseOK) ? " was contacted successfully." : 
087              " NOT AVAILABLE. This service will not run correctly."));
088        server.logger.warning("DailyProjectData (Version " + getVersion() + ") now running.");
089        server.component.start();
090        disableRestletLogging();
091        return server;
092      }
093    
094      /**
095       * Disable all loggers from com.noelios and org.restlet. 
096       */
097      private static void disableRestletLogging() {
098        LogManager logManager = LogManager.getLogManager();
099        for (Enumeration e = logManager.getLoggerNames(); e.hasMoreElements() ;) {
100          String logName = e.nextElement().toString();
101          if (logName.startsWith("com.noelios") ||
102              logName.startsWith("org.restlet")) {
103            logManager.getLogger(logName).setLevel(Level.OFF);
104          }
105        }
106      }
107      
108      /**
109       * Starts up the web service.  Control-c to exit. 
110       * @param args Ignored. 
111       * @throws Exception if problems occur.
112       */
113      public static void main(final String[] args) throws Exception {
114        Server.newInstance();
115      }
116    
117      /**
118       * Dispatch to the specific DailyProjectData resource based upon the URI.
119       * We will authenticate all requests.
120       * @return The router Restlet.
121       */
122      @Override
123      public Restlet createRoot() {
124        // First, create a Router that will have a Guard placed in front of it so that this Router's
125        // requests will require authentication.
126        Router authRouter = new Router(getContext());
127        authRouter.attach("/devtime/{user}/{project}/{timestamp}", DevTimeResource.class);
128        // Here's the Guard that we will place in front of authRouter.
129        Guard guard = new Authenticator(getContext(), 
130            this.getServerProperties().get(SENSORBASE_HOST_KEY));
131        guard.setNext(authRouter);
132        return guard;
133      }

137       * Returns the version associated with this Package, if available from the jar file manifest.
138       * If not being run from a jar file, then returns "Development".

142        String version = 
143          Package.getPackage("org.hackystat.dailyprojectdata.server").getImplementationVersion();
144        return (version == null) ? "Development" : version; 
145      }

162      }
