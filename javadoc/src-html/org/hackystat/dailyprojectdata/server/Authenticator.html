001    package org.hackystat.dailyprojectdata.server;
002    
003    import java.util.HashMap;
004    
005    import java.util.Map;
006    
007    import org.hackystat.sensorbase.client.SensorBaseClient;
008    import org.restlet.Context;
009    import org.restlet.Guard;
010    import org.restlet.data.ChallengeScheme;
011    
012    /**
013     * Performs authentication of each HTTP request using HTTP Basic authentication. 
014     * Checks user/password credentials by pinging SensorBase, then caching authentic 
015     * user/password combinations.  If a cached user/password combo does not match the 
016     * current user/password combo, then the SensorBase is pinged again (because maybe
017     * the user has changed their password recently). 
018     * <p>
019     * Because DPD resources will always want to communicate with the underlying SensorBase, this 
020     * Authenticator also creates a map of user-to-SensorBaseClient instances which can be retrieved
021     * from the server context.  This keeps the user password info in this class, while making the 
022     * SensorBaseClient instance available to the service. 
023     * @author Philip Johnson
024     */
025    public class Authenticator extends Guard {
026      
027      /** A map containing previously verified credentials. */
028      private Map<String, String> credentials = new HashMap<String, String>();
029      
030      /** A map containing SensorBaseClient instances, one per credentialed user. */
031      private Map<String, SensorBaseClient> userClientMap = new HashMap<String, SensorBaseClient>();
032      
033      /** The sensorbase host, such as "http://localhost:9876/sensorbase/" */
034      private String sensorBaseHost;
035      
036      /** The key to be used to retrieve the sensorbaseclient map from the server context. */
037      public static final String AUTHENTICATOR_SENSORBASECLIENTS_KEY = "authenticator.clients";
038    
039      /**
040       * Initializes this Guard to do HTTP Basic authentication.
041       * Puts the credentials map in the server context so that Resources can get the 
042       * password associated with the uriUser for their own invocations to the SensorBase. 
043       * @param context The server context.
044       * @param sensorBaseHost The host, such as 'http://localhost:9876/sensorbase/'.
045       */
046      public Authenticator (Context context, String sensorBaseHost) {
047        super(context, ChallengeScheme.HTTP_BASIC,  "DailyProjectData");
048        this.sensorBaseHost = sensorBaseHost;
049        context.getAttributes().put(AUTHENTICATOR_SENSORBASECLIENTS_KEY, userClientMap);
050      }
051      
052      /**
053       * Returns true if the passed credentials are OK.
054       * @param identifier The account name.
055       * @param secretCharArray The password. 
056       * @return If the credentials are valid.
057       */
058      @Override protected boolean checkSecret(String identifier, char[] secretCharArray) {
059        String secret = new String(secretCharArray);
060        // Return true if the user/password credentials are in the cache. 
061        if (credentials.containsKey(identifier) &&
062            secret.equals(credentials.get(identifier))) {
063          return true;
064        }
065        // Otherwise we check the credentials with the SensorBase.
066        boolean isRegistered = SensorBaseClient.isRegistered(sensorBaseHost, identifier, secret);
067        if (isRegistered) {
068          // Credentials are good, so save them and create a sensorbase client for this user. 
069          credentials.put(identifier, secret);
070          userClientMap.put(identifier, new SensorBaseClient(sensorBaseHost, identifier, secret));
071        }
072        return isRegistered;
073      }
074    }
