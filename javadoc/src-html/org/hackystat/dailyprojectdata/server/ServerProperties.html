001    package org.hackystat.dailyprojectdata.server;
002    
003    import java.io.FileInputStream;
004    import java.io.IOException;
005    import java.util.Properties;
006    
007    /**
008     * Provides access to the values stored in the dailyprojectdata.properties file. 
009     * @author Philip Johnson
010     */
011    public class ServerProperties {
012      
013      /** The sensorbase host. */
014      public static final String SENSORBASE_HOST_KEY =  "dailyprojectdata.sensorbase.host";
015      /** The dailyprojectdata hostname key. */
016      public static final String HOSTNAME_KEY =        "dailyprojectdata.hostname";
017      /** The dailyprojectdata context root. */
018      public static final String CONTEXT_ROOT_KEY =     "dailyprojectdata.context.root";
019      /** The logging level key. */
020      public static final String LOGGING_LEVEL_KEY =   "dailyprojectdata.logging.level";
021      /** The dailyprojectdata port key. */
022      public static final String PORT_KEY =            "dailyprojectdata.port";
023      /** The XML directory key. */
024      public static final String XML_DIR_KEY =         "dailyprojectdata.xml.dir";
025      
026      /** Where we store the properties. */
027      private Properties properties; 
028      
029      /**
030       * Creates a new ServerProperties instance. 
031       * Prints an error to the console if problems occur on loading. 
032       */
033      public ServerProperties() {
034        try {
035          initializeProperties();
036        }
037        catch (Exception e) {
038          System.out.println("Error initializing DPD server properties.");
039        }
040      }
041      
042      /**
043       * Reads in the properties in ~/.hackystat/dailyprojectdata/dailyprojectdata.properties 
044       * if this file exists, and provides default values for all properties. .
045       * @throws Exception if errors occur.
046       */
047      private void initializeProperties () throws Exception {
048        String userHome = System.getProperty("user.home");
049        String userDir = System.getProperty("user.dir");
050        String propFile = userHome + "/.hackystat/dailyprojectdata/dailyprojectdata.properties";
051        this.properties = new Properties();
052        // Set defaults
053        properties.setProperty(SENSORBASE_HOST_KEY, "http://localhost:9876/sensorbase/");
054        properties.setProperty(HOSTNAME_KEY, "localhost");
055        properties.setProperty(PORT_KEY, "9877");
056        properties.setProperty(CONTEXT_ROOT_KEY, "dailyprojectdata");
057        properties.setProperty(LOGGING_LEVEL_KEY, "INFO");
058        properties.setProperty(XML_DIR_KEY, userDir + "/xml");
059        FileInputStream stream = null;
060        try {
061          stream = new FileInputStream(propFile);
062          properties.load(stream);
063          System.out.println("Loading DailyProjectData properties from: " + propFile);
064        }
065        catch (IOException e) {
066          System.out.println(propFile + " not found. Using default dailyprojectdata properties.");
067        }
068        finally {
069          if (stream != null) {
070            stream.close();
071          }
072        }
073        // Now add to System properties.
074        Properties systemProperties = System.getProperties();
075        systemProperties.putAll(properties);
076        System.setProperties(systemProperties);
077      }
078    
079      /**
080       * Prints all of the DPD settings to the logger.
081       * @param server The DPD server.   
082       */
083      public void echoProperties(Server server) {
084        String cr = System.getProperty("line.separator"); 
085        String eq = " = ";
086        String pad = "                ";
087        String propertyInfo = "DailyProjectData Properties:" + cr +
088          pad + SENSORBASE_HOST_KEY   + eq + get(SENSORBASE_HOST_KEY) + cr +
089          pad + HOSTNAME_KEY      + eq + get(HOSTNAME_KEY) + cr +
090          pad + CONTEXT_ROOT_KEY  + eq + get(CONTEXT_ROOT_KEY) + cr +
091          pad + LOGGING_LEVEL_KEY + eq + get(LOGGING_LEVEL_KEY) + cr +
092          pad + PORT_KEY          + eq + get(PORT_KEY);
093        server.getLogger().info(propertyInfo);
094      }
095      
096      /**
097       * Returns the value of the Server Property specified by the key.
098       * @param key Should be one of the public static final strings in this class.
099       * @return The value of the key, or null if not found.
100       */
101      public String get(String key) {
102        return this.properties.getProperty(key);
103      }
104      
105      /**
106       * Returns the fully qualified host name, such as "http://localhost:9877/dailyprojectdata/".
107       * @return The fully qualified host name.
108       */
109      public String getFullHost() {
110        return "http://" + get(HOSTNAME_KEY) + ":" + get(PORT_KEY) + "/" + get(CONTEXT_ROOT_KEY) + "/";
111      }
112    }
