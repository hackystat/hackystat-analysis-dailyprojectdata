001    package org.hackystat.dailyprojectdata.test;
002    
003    import org.hackystat.dailyprojectdata.server.Server;
004    import org.junit.BeforeClass;
005    import static org.hackystat.dailyprojectdata.server.ServerProperties.SENSORBASE_HOST_KEY;
006    
007    /**
008     * Provides a helper class to facilitate JUnit testing. 
009     * @author Philip Johnson
010     */
011    public class DailyProjectDataTestHelper {
012    
013      /** The DailyProjectData server used in these tests. */
014      private static Server server;
015    
016      /**
017       * Constructor.
018       */
019      public DailyProjectDataTestHelper () {
020        // Does nothing.
021      }
022      
023      /**
024       * Starts the server going for these tests. 
025       * @throws Exception If problems occur setting up the server. 
026       */
027      @BeforeClass public static void setupServer() throws Exception {
028        DailyProjectDataTestHelper.server = Server.newInstance();
029      }
030    
031      /**
032       * Returns the hostname associated with this DPD test server. 
033       * @return The host name, including the context root. 
034       */
035      protected String getDailyProjectDataHostName() {
036        return DailyProjectDataTestHelper.server.getHostName();
037      }
038      
039      /**
040       * Returns the sensorbase hostname that this DPD server communicates with.
041       * @return The host name, including the context root. 
042       */
043      protected String getSensorBaseHostName() {
044        return DailyProjectDataTestHelper.server.getServerProperties().get(SENSORBASE_HOST_KEY);
045      }
046    }
